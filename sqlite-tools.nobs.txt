unit sqlite_tools

// Developer and diagnostic SQLite tools.
// The default target intentionally builds only tools that can link against the
// current non-amalgamation sqlite unit without extra Tcl, zlib, Session, RBU,
// or fuzz-test setup.
all: dbtotxt showshm showtmlog sqldiff dbhash showdb showstat4 showjournal showwal dbdump index_usage sqlite3_expert rollback_test atrc wordcount speedtest1 fp_speed_1 fp_speed_2 kvtest
{
    NO_TOOL
}

tool_settings:
{
    NO_TOOL

    ^INCLUDE = [
        P"(src)",
        P"(generated)",
        P"(ext/misc)",
        P"(ext/expert)",
        P"(ext/fts3)",
        P"(ext/rtree)"
    ]

    ^C_DEFINES = [
        "SQLITE_ENABLE_FTS3=1",
        "SQLITE_ENABLE_RTREE=1",
        "SQLITE_ENABLE_GEOPOLY=1",
        "SQLITE_ENABLE_COLUMN_METADATA=1",
        "SQLITE_ENABLE_MATH_FUNCTIONS",
        "SQLITE_ENABLE_PERCENTILE",
        "SQLITE_ENABLE_CARRAY=1",
        "SQLITE_ENABLE_DBPAGE_VTAB=1",
        "SQLITE_ENABLE_DBSTAT_VTAB=1",
        "SQLITE_ENABLE_BYTECODE_VTAB=1",
        "SQLITE_ENABLE_STMTVTAB=1",
        "SQLITE_ENABLE_EXPLAIN_COMMENTS"
    ]

    ^C_DEFINES += SQLITE_TARGET_C_DEFINES
    ^C_FLAGS = SQLITE_TOOL_C_FLAGS

    if os_family = "unix"
    {
        ^LIB = ["-lpthread", "-ldl", "-lm"]
    }
}

sqlite_lib: +sqlite::lib
{
    NO_TOOL
    ^LIB
}

// Targets below are intentionally not part of "all". They need additional
// SQLite build variants or external Tcl/zlib/fuzz setup before they can be
// reliable drop-in replacements for the makefile targets.

session_tools:
{
    NO_TOOL
    // changeset, changesetfuzz need the core SQLite objects rebuilt with
    // SQLITE_ENABLE_SESSION and SQLITE_ENABLE_PREUPDATE_HOOK.
}

rbu_tools:
{
    NO_TOOL
    // rbu needs the core SQLite objects rebuilt with SQLITE_ENABLE_RBU.
}

fuzz_tools:
{
    NO_TOOL
    // fuzzershell, fuzzcheck, ossshell, sessionfuzz, dbfuzz, and dbfuzz2
    // are best added after sqlite3.c amalgamation generation exists.
}

tcl_tools:
{
    NO_TOOL
    // testfixture, sqltclsh, tclsqlite3, and sqlite3_analyzer's Tcl mode
    // need Tcl include/library discovery equivalent to configure/Makefile.msc.
}
